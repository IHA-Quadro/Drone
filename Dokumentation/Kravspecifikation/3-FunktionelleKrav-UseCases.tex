\subsection{Use Case 1}

\begin{UseCase}
\UCnumber{1}{System initialisering.}
\UCgoal{Drone er klar til at flyve.}
\UCinit{Use Case initieres af bruger.}
\UCslutSuc{Propeller spinner med indbyrdes ens hastighed. }
\UCslutUnSuc{Drone bipper ét kort bip med 2-3 sekunders mellemrum.}

\UCnormFor{Normalforløb}
	\begin{normFor}
	\item Bruger kobler batteriets strømkabel til dronens strømkabel.
	\item Bruger venter på sekvensen af bips: ét kort, pause, tre korte, pause, ét langt.
	\item[]	\textit{Undtagelse: Opstartssekvens.}
	\item Dronens propeller begynder at spinde med indbyrdes ens hastighed.
	\end{normFor} 

\UCnormFor{Undtagelser}
	\begin{normFor}
	\item \textit{Undtagelse: Opstartssekvens.}
	\begin{itemize}
	\item Drone bipper én gang hver 2-3 sekund, hvilket skyldes der ikke modtages 11.1 V fra batteriet.
	\end{itemize}
	\end{normFor} 
\end{UseCase}



\subsection{Use Case 2}

\begin{UseCase}
\UCnumber{2}{Programvalg}
\UCgoal{Dronen udfører valgte program.}
\UCinit{Use Case initieres af bruger}
\UCstartb{Dronens batteri er opladt og Use Case 1 er kørt.} 
\UCslutSuc{Dronen udførte valgte program.}
\UCslutUnSuc{Dronen reagerer ikke på tryk eller dronens propeller spinner ikke længere}

\UCnormFor{Normalforløb 1: \\ Let}
	\begin{normFor}
	\item Brugeren trykker på knap for let.
	\item Dronens propeller spinner hurtigere og letter dronen.
	\item[] \textit{Undtagelse: Drone reagerer ikke}
	\item Dronen holder sig stabilt i 100 cm højde $\pm$ 10 cm \fxnote{Test og ret dette tal}.
	\end{normFor} 
	
\UCnormFor{Normalforløb 2: \\ Autoland}
	\begin{normFor}
	\item Bruger trykker på autoland-knap
	\item Drone sænker sig langsomt mod jorden, lander og slukker propellerne.
	\item[] \textit{Undtagelse: Drone reagerer ikke}
	\end{normFor} 
	
\UCnormFor{Normalforløb 3: \\ Stop} 
	\begin{normFor}
	\item Bruger trykker på stop-knap
	\item Drone slukker for strømmen til sine motorer.
	\item[] \textit{Undtagelse: Drone reagerer ikke}
	\end{normFor} 
	
\UCnormFor{Normalforløb 4: \\ Sænk spind-hastighed} 
	\begin{normFor}
	\item Bruger trykker på ned-knap
	\item Drone sænker motorernes rotationshastighed.
	\item[] \textit{Undtagelse: Drone reagerer ikke}
	\end{normFor} 
	
\UCnormFor{Normalforløb 5: \\ Øg spind-hastighed} 
	\begin{normFor}
	\item Bruger trykker på op-knap
	\item Drone forøger motorernes rotationshastighed.
	\item[] \textit{Undtagelse: Drone reagerer ikke}
	\end{normFor} 
	
\UCnormFor{Normalforløb 6: \\ Roter til højre} 
	\begin{normFor}
	\item Bruger trykker på højre-knap
	\item Drone drejer til højre.
	\item[] \textit{Undtagelse: Drone reagerer ikke}
	\end{normFor} 
	
\UCnormFor{Normalforløb 7: \\ Roter til venstre} 
	\begin{normFor}
	\item Bruger trykker på venstre-knap
	\item Drone drejer til venstre.
	\item[] \textit{Undtagelse: Drone reagerer ikke}
	\end{normFor} 
	
\UCnormFor{Normalforløb 8: \\ Flyv fremad} 
	\begin{normFor}
	\item Bruger trykker på fremad-knap
	\item Drone flyver fremad.
	\item[] \textit{Undtagelse: Drone reagerer ikke}
	\end{normFor} 

\UCnormFor{Undtagelser}
	\begin{normFor}
	\item[2.] \textit{Undtagelse: Drone reagerer ikke}
		\begin{itemize}
		\item Dronen fortsætter sit tidligere program.
		\end{itemize}
	\end{normFor} 
	
\end{UseCase}




\subsection{Use Case 3}
\begin{UseCase}
\UCnumber{3}{Sonaradvarsel.}
\UCgoal{Drone vender sig væk fra registreret udfordring.}
\UCinit{Use Case initieres af drone.}
\UCstartb{UC 1 er kørt og dronen er i luften} 
\UCslutSuc{Dronens kan ikke længere se en udfordring for tæt på.}
\UCslutUnSuc{Dronen flyver ind i objekt.}

\UCnormFor{Normalforløb}
	\begin{normFor}
	\item Drone registrerer objekt inden for 100 cm afstand $\pm$ 10 cm\fxnote{Test og ret}.
	\item Drone stopper i luften og roterer for at undgå dette.
	\item[] \textit{Undtagelse: Drone stopper ikke}
	\end{normFor} 

\UCnormFor{Undtagelser}
	\begin{normFor}
	\item[2.] \textit{Undtagelse: Drone stopper ikke}
		\begin{itemize}
		\item Drone flyver ind i objekt.
		\end{itemize}
	\end{normFor} 
	
\end{UseCase}


\subsection{Use Case 4}
\begin{UseCase}
\UCnumber{4}{Maks. spin stop.}
\UCgoal{Dronens propeller stopper.}
\UCinit{Use Case initieres af drone.}
\UCstartb{UC 1 er kørt og dronen er i luften.} 
\UCslutSuc{Dronen er på jorden.}
\UCslutUnSuc{Ingen.}

\UCnormFor{Normalforløb}
	\begin{normFor}
	\item Dronen registrerer én af sine motorer flyver med for høj hastighed.
	\item Drone slukker helt for sine motorer.
	\end{normFor} 
\end{UseCase}

\item Dronen flyver ikke nødvendigvis opad ved tryk på "op".
\end{itemize}
